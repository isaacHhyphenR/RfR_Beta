#	Triggers to control if diplomatic actions are enabled:
#		Format:
#
#		<diplomatic action tag>_enabled_trigger = {
#			<trigger contents>
#		}
#
#		ROOT is the country initiating the diplomatic action.
#		FROM is the recipient or target of the diplomatic action.
#
#		Note:  For cleaner tooltips, it is recommended that you use
#			if triggers within the enable triggers, and a single
#			custom_trigger_tooltip trigger within each if trigger.
#
#		Example: (Germany is prevented from declaring war on the Soviet Union.)
#
#		DIPLOMACY_WAR_ENABLE_TRIGGER = {
#			if = {
#				limit = {
#					tag = GER
#				}
#				custom_trigger_tooltip = {
#					tooltip = PREVENT_GER_WAR_ON_SOV
#					FROM = {
#						NOT = {
#							tag = SOV
#						}
#					}
#				}
#			}
#		}

# This specifically overrides the rule check as to whether or not you can guarantee other ideologies
DIPLOMACY_GUARANTEE_ENABLE_TRIGGER_OVERRIDES_GAME = {
	NOT = {
		has_game_rule = {
			rule = allow_guarantees
			option = LIMITED
		}
	}
}

DIPLOMACY_GUARANTEE_ENABLE_TRIGGER = {
	if = {
		limit = {
			has_game_rule = {
				rule = allow_guarantees
				option = ALWAYS_FREE
			}
		}
		always = yes
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_guarantees
				option = LIMITED
			}
		}
		always = yes # Handled by game logic.
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_guarantees
				option = SAME_IDEOLOGY
			}
		}
		custom_trigger_tooltip = {
			tooltip = RULE_ALLOW_GUARANTEES_SAME_IDEOLOGY_TOOLTIP
			has_government = FROM
		}
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_guarantees
				option = BLOCKED
			}
		}
		custom_trigger_tooltip = {
			tooltip = RULE_ALLOW_GUARANTEES_BLOCKED_TOOLTIP
			always = no
		}
	}
}

DIPLOMACY_GENERATE_WARGOAL_ENABLE_TRIGGER = {
	#melkorite
	if = {
		limit = {
			has_government = melkorite
		}
		custom_trigger_tooltip = {
			tooltip = diplomacy_tooltip_melkorite_hasnt_unleashed_darkness
			any_country = { check_variable = { modifier@melkorite_can_justify > 0 } }
		}
	}
	##Rules
	if = {
		limit = {
			has_game_rule = {
				rule = allow_wargoals
				option = ALWAYS_FREE
			}
		}
		always = yes
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_wargoals
				option = LIMITED
			}
		}
		always = yes
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_wargoals
				option = FREE_25
			}
		}
		threat>0.24
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_wargoals
				option = FREE_50
			}
		}
		threat>0.49
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_wargoals
				option = FREE_75
			}
		}
		threat>0.74
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_wargoals
				option = FREE_100
			}
		}
		threat>0.99
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_wargoals
				option = FOCUSES_ONLY
			}
		}
		custom_trigger_tooltip = {
			tooltip = RULE_WARGOALS_BLOCKED_TT
			always = no
		}
	}
}

DIPLOMACY_JOIN_FACTION_ENABLE_TRIGGER = {
	if = { ##non-gwaithurium tags cannot join gwaithurium led faction
		limit = {
			var:FROM.faction_leader = { is_greater_enedwaith_tag = yes }
		}
		custom_trigger_tooltip = {
			tooltip = cannot_join_gwaithurium_faction_tooltip
			ROOT = {
				is_greater_enedwaith_tag = yes
			}
		}
	}
	if = { ##Melkorite/Conquest cannot join Isolationist/Freedom faction.
		limit = {
			var:FROM.faction_leader = {
				OR = {
					has_government = isolationism
					has_government = freedom
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = cannot_join_opposing_ideology_faction
			ROOT = {
				NOT = {
					OR = {
						has_government = melkorite
						has_government = conquest
					}
				}
			}
		}
	}
	else_if = { ##Isolationist/Freedom cannot join Melkorite/Conquest faction.
		limit = {
			var:FROM.faction_leader = {
				OR = {
					has_government = melkorite
					has_government = conquest
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = cannot_join_opposing_ideology_faction
			ROOT = {
				NOT = {
					OR = {
						has_government = isolationism
						has_government = freedom
					}
				}
			}
		}
	}
	if = { ##Only celduin trade countries can join celduin trade league
		limit = {
			var:FROM.faction_leader = { has_idea = head_of_celduin_trade_league }
		}
		custom_trigger_tooltip = {
			tooltip = celduin_trade_league_only_celduin_states
			ROOT = {
				allowed_celduin_trade = yes
			}
		}
	}
	else_if = { ##only republics can join The Revolution
		limit = {
			var:FROM.faction_leader = { has_dynamic_modifier = { modifier = DOR_home_of_the_revolution_modifier } }
		}
		custom_trigger_tooltip = {
			tooltip = the_revolution_only_republics
			ROOT = {
				is_republic = yes
			}
		}
	}
}

lar_local_recruitment = {
###################unlocks decisions to recruit in easterling countries
	recruit_easterling_agent = {
		icon = GFX_decision_SWI_expand_covert_operations
		visible = {
			capital_scope = {
				NOT = {
					is_on_continent = easterling_race
				}
			}
			OR = {
				any_owned_state = {
					is_on_continent = easterling_race
				}
				any_allied_country = {
					any_owned_state = {
						is_on_continent = easterling_race
					}
				}
				any_of_scopes = {
					array = subjects
					any_owned_state = {
						is_on_continent = easterling_race
					}
				}
			}
		}

		fire_only_once = yes
		cost = 50

		modifier = {
			civilian_factory_use = 3
		}

		days_remove = 90

		remove_effect = {
			set_country_flag = easterling_recruitment_unlocked
		}

		ai_will_do = {
			base = 0
		}
	}

	recruit_in_easterling_state = {
		icon = onmap_recruit_operative
		state_target = yes
		target_trigger = {
			FROM = {
				is_on_continent = easterling_race
				OR = {
					AND = {
						has_variable = historical_capital_for_country
						var:historical_capital_for_country = {
							exists = yes
							has_capitulated = yes
							NOT = { tag = ROOT }
						}
					}
					AND = {
						is_capital = yes
						CONTROLLER = {
							NOT = {
								tag = ROOT
							}
						}
					}
				}
			}
		}
		visible = {
			OR = {
				has_country_flag = eriador_recruitment_unlocked
				capital_scope = { is_on_continent = easterling_race }
			}
			NOT = { has_country_flag = easterling_recruitment_in_process }
		}

		fire_only_once = no
		cost = 0

		days_re_enable = 180
		on_map_mode = map_only
		complete_effect = {
			set_country_flag = { flag = easterling_recruitment_in_process days = 180 value = 1 }
			if = {
				limit = {
					num_of_operatives < num_operative_slots
				}
				custom_effect_tooltip = recruit_in_state_tt
			}
			else = {
				custom_effect_tooltip = recruit_in_state_to_pool_tt
			}
			hidden_effect = {
				create_operative_leader = {
					bypass_recruitment = yes
					nationalities = { FROM.OWNER }
					portrait_tag_override = FROM.OWNER
				}
			}
		}

		ai_will_do = {
			base = 0
		}
	}
